In Problems 1 - 18, use \hyperlink{def:Laplace 7.1.1}{\textbf{Definition 7.1.1}} to find $\mathscr{L} \{ f(t) \}$.

\begin{boxeddefinition}[7.1.1] \hypertarget{def:Laplace 7.1.1}{}
  \ \newline
  Let $f$ be a function defined for $t \geq 0$. Then the integral
    \begin{equation}
        \mathscr{L}\{f(t)\} = \int_{0}^{\infty} e^{-st} f(t) \, dt
    \end{equation}
  is said to be the \textbf{Laplace transform} of $f$, provided that the integral converges.  
\end{boxeddefinition}

  \subsection*{Problem 7.1.4}
  \addcontentsline{toc}{subsection}{Problem 7.1.4}
  \vspace{-0.5cm}

  \begin{flalign*}
    & \text{Given: } f(t) =
      \begin{cases}
          2t+1    & 0 \leq t \leq 1  \\[2mm]
          0       & t \geq 1
      \end{cases} & \\
    & \text{Find:  } \mathscr{L}\{f(t)\} &   
  \end{flalign*}
  \begin{solution}
    Using \hyperlink{def:Laplace 7.1.1}{\textbf{Definition 7.1.1}}, we split the integral 
    to conform with the piece $f(t)$ function.

    \begin{align*}
      \mathscr{L}\{f(t)\} &= \int_{0}^{\infty} e^{-st} f(t) \, dt \\
      \mathscr{L}\{f(t)\} &= \int_{0}^{1} e^{-st} f(t) \, dt + \int_{1}^{\infty} e^{-st} f(t) \, dt \\
      \mathscr{L}\{f(t)\} &= \int_{0}^{1} e^{-st} \left(2t+1\right) \, dt + \cancelto{0}{\int_{1}^{\infty} e^{-st} (0) \, dt} \\
      \mathscr{L}\{f(t)\} &= \int_{1}^{\infty} e^{-st} \left(2t+1\right) \, dt \\
      \mathscr{L}\{f(t)\} &= \left[-\frac{\left(2t+1\right)}{s}e^{-st}-\frac{2}{s^{2}}e^{-st} \right]_{t=0}^{t=1} 
    \end{align*}

    \begin{align*}
                       &  \mathscr{L}\{f(t)\} = \left(-\frac{\left(2\left(1\right)+1\right)}{s}e^{-s\left(1\right)}-\frac{2}{s^{2}}e^{-s\left(1\right)}\right)-\left(-\frac{\left(2\left(0\right)+1\right)}{s}e^{-s\left(0\right)}-\frac{2}{s^{2}}e^{-s\left(0\right)}\right) \\
                       &  \mathscr{L}\{f(t)\} = \left(-\frac{3}{s}e^{-s}-\frac{2}{s^{2}}e^{-s}\right)+\left(\frac{1}{s}+\frac{2}{s^{2}}\right) \\
                       &  \mathscr{L}\{f(t)\} = \frac{1}{s}\left(1-3e^{-s}\right)+\frac{2}{s^{2}}\left(1-e^{-s}\right) \\
      \Longrightarrow  &  \boxed{\mathscr{L}\{f(t)\} = \frac{1}{s}\left(1-3e^{-s}\right)+\frac{2}{s^{2}}\left(1-e^{-s}\right), \ s > 0}
    \end{align*}

  \end{solution}

  \begin{flalign*}
      & \text{Given: } f(t) = e^{-2t-5} & \\
      & \text{Find:  } \mathscr{L}\{f(t)\} &
  \end{flalign*}
  \begin{solution}

    to conform with the piece $f(t)$ function.

    \begin{align*}
      \mathscr{L}\{f(t)\} &= \int_{0}^{\infty} e^{-st} f(t) \, dt \\
      \mathscr{L}\{f(t)\} &= \int_{0}^{\infty} e^{-st} \left(e^{-2t-5}\right) \, dt  \\
      \mathscr{L}\{f(t)\} &= \int_{0}^{\infty}e^{\left(-t\left(s+2\right)-5\right)} \, dt \\
      \mathscr{L}\{f(t)\} &= e^{-5}\int_{0}^{\infty}e^{-t\left(s+2\right)} \, dt \\
      \mathscr{L}\{f(t)\} &= \left[ -\frac{e^{-5}}{s+2}\left(e^{-t}\right) \right]_{t=0}^{t=\infty} \\
      \mathscr{L}\{f(t)\} &=  -\frac{e^{-5}}{s+2} \left[ e^{-t} \right]_{t=0}^{t=\infty}
    \end{align*}

    \begin{align*}
                       &  \mathscr{L}\{f(t)\} = -\frac{e^{-5}}{s+2}\left(\lim_{t \to \infty} \left(e^{-t} \right) -e^{-\left(0\right)}\right) \\
                       &  \mathscr{L}\{f(t)\} = -\frac{e^{-5}}{s+2}\left(0-1\right) \\
      \Longrightarrow  &  \boxed{\mathscr{L}\{f(t)\} = \frac{e^{-5}}{s+2}, \ s > -2}
    \end{align*}
  \end{solution}

\begin{flalign*}
    & \text{Given: } f(t) = \left(2t-1\right)^{3} & \\
    & \text{Find:  } \mathscr{L}\{f(t)\} &
\end{flalign*}
\begin{solution}
  Using the table from \hyperlink{def:Theorem 7.1.1}{\textbf{Theorem 7.1.1}}, we can use 
  the linear property of the Laplacian and evaluate. Here we are using the power rule.

  \begin{align*}
    \mathscr{L}\{f(t)\} &= \mathscr{L}\left\{\left(2t-1\right)^{3} \right\} \\
                        &= \mathscr{L}\left\{8t^{3}-3\left(2t\right)^{2}\left(1\right)+3\left(2t\right)\left(1\right)^{2}-\left(1\right)^{3} \right\} \\
                        &= \mathscr{L}\left\{8t^{3}-12t^{2}+6t-1 \right\} \\
                        &= \mathscr{L}\left\{8t^{3} \right\} + \mathscr{L}\left\{-12t^{2} \right\} + \mathscr{L}\left\{6t \right\} - \mathscr{L}\left\{1 \right\} \\
                        &= 8\mathscr{L}\left\{t^{3} \right\} - 12\mathscr{L}\left\{t^{2} \right\} + 6\mathscr{L}\left\{t \right\} - \mathscr{L}\left\{1 \right\} \\
                        &= 8\left(\frac{3!}{s^{3+1}}\right) - 12\left(\frac{2!}{s^{2+1}}\right) + 6\left(\frac{1!}{s^{1+1}}\right) - \left(\frac{1}{s}\right) \\
    \mathscr{L}\{f(t)\} &= \frac{48}{s^{4}}-\frac{24}{s^{3}}+\frac{6}{s^{2}}-\frac{1}{s}
  \end{align*}

  \begin{equation*}
    \Longrightarrow \boxed{\mathscr{L}\{f(t)\} = \frac{48}{s^{4}}-\frac{24}{s^{3}}+\frac{6}{s^{2}}-\frac{1}{s}, \ s > 0}
  \end{equation*}

\end{solution}

\begin{boxedtheorem}[7.2.2] \hypertarget{def:Theorem 7.2.2}{}
  \ \newline
  \textbf{Transform of a Derivative} \newline
  If $f, f', \dots, f^{(n-1)}$ are continuous on $[0, \infty)$ and are of exponential order
  and if $f^{(n)}(t)$ is piecewise continuous on $[0, \infty)$, then

  \vspace{-0.5cm}
  \begin{align}
    \mathscr{L}\{f^{(n)}(t)\} &= s^n F(s) - s^{n-1} f(0) - s^{n-2} f'(0) - \dots - f^{(n-1)}(0) \\
                              &= s^n F(s) -\sum_{k=1}^{n}s^{\left(k-1\right)}f^{\left(n-k\right)}(0),
  \end{align}

  where $F(s) = \mathscr{L}\{f(t)\}$. 
\end{boxedtheorem}

  \end{flalign*}
  \begin{solution}
  The main process to apply the Laplace transform, using \hyperlink{def:Theorem 7.1.1}{\textbf{Theorem 7.1.1}} and \hyperlink{def:Theorem 7.2.2}{\textbf{Theorem 7.2.2}}, over the entire equation,
  isolate $Y(s)$, and apply the inverse Laplace transform using \hyperlink{def:Theorem 7.2.1}{\textbf{Theorem 7.2.1}}.

  \begin{align*}
                                    & 2\frac{dy}{dt}+y  = 0 \\
    \mathscr{L} \hspace{-0.10cm}: \ & 2\frac{dy}{dt}+y \\
                    \Longrightarrow & 2\left(-y(0)+sY(s)\right)+Y(s) = 0 
  \end{align*}
  Rearrange and isolate $Y(s)$
  \begin{align*}
    Y(s)\left(2s+1\right) &= 2y(0) \\
                     Y(s) &= \frac{2y(0)}{\left(2s+1\right)} \\
                     Y(s) &= \frac{2\left(-3\right)}{\left(2s+1\right)} \\
                     Y(s) &= -\frac{6}{\left(2s+1\right)} \\
                     Y(s) &= -\frac{6}{2\left(s+\frac{1}{2}\right)} \\
                     Y(s) &= -\frac{3}{\left(s+\frac{1}{2}\right)}
  \end{align*}
  Apply the inverse Laplace transform.
  \begin{align*}
     \mathscr{L}^{-1} \hspace{-0.10cm}: \ & Y(s) = -\frac{3}{\left(s+\frac{1}{2}\right)} \\
                                        \Longrightarrow  & \boxed{y(t) = -3e^{-\frac{1}{2}t}}
  \end{align*}

  \end{solution}

  \subsubsection*{Problem 7.2.40}
  \addcontentsline{toc}{subsubsection}{Problem 7.2.40}
  \vspace{-0.5cm}

  \begin{flalign*}
      & \text{Given: } y''-4y'=6e^{3t}-3e^{-t}, \ y(0) = 1, \ y'(0) = -1 & \\
      & \text{Find:  } y(t) &
  \end{flalign*}
  \begin{solution}
  The main process to apply the Laplace transform, using \hyperlink{def:Theorem 7.1.1}{\textbf{Theorem 7.1.1}} and \hyperlink{def:Theorem 7.2.2}{\textbf{Theorem 7.2.2}}, over the entire equation,
  isolate $Y(s)$, and apply the inverse Laplace transform using \hyperlink{def:Theorem 7.2.1}{\textbf{Theorem 7.2.1}}.

  \begin{align*}
                                    & y''-4y'=6e^{3t}-3e^{-t} \\
    \mathscr{L} \hspace{-0.10cm}: \ & y''-4y'=6e^{3t}-3e^{-t} \\
                    \Longrightarrow & \left(-y'(0)-sy(0)+s^{2}Y(s)\right)-4\left(-y(0)+sY(s)\right)=\frac{6}{s-3}-\frac{3}{s+1}
  \end{align*}
  Rearrange and isolate $Y(s)$.
  \begin{align*}
          -y'(0)-sy(0)+4y(0)-4s+s^{2}Y(s)-4sY(s) &= \frac{6}{s-3}-\frac{3}{s+1} \\
          Y(s)\left(s^{2}-4s\right) &= \frac{6}{s-3}-\frac{3}{s+1}+y'(0)+y(0)\left(s-4\right) \\
          Y(s)\left(s^{2}-4s\right) &= \frac{6}{s-3}-\frac{3}{s+1}+\left(-1\right)+\left(1\right)\left(s-4\right) \\
          Y(s)\left(\left(s\right)\left(s-4\right)\right) &= \frac{6}{s-3}-\frac{3}{s+1}+\left(s-5\right) \\
          Y(s) &= \frac{6}{s\left(s-3\right)\left(s-4\right)}-\frac{3}{s\left(s+1\right)\left(s-4\right)}+\frac{s-5}{s\left(s-4\right)} \\
          Y(s) &= \frac{A}{s}+\frac{B}{s-3}+\frac{C}{s-4}+\frac{D}{s+1}
  \end{align*}
  Using Heaviside cover up method.
  \begin{align*}
    & A=\frac{5}{2}, \ B=-2, \ C=\frac{11}{10}, \ D=-\frac{3}{5} \\
                \Longrightarrow Y(s) &= \frac{\frac{5}{2}}{s}+\frac{-2}{s-3}+\frac{\frac{11}{10}}{s-4}+\frac{-\frac{3}{5}}{s+1}
  \end{align*}
  Apply the inverse Laplace transform.
  \begin{align*}
    \mathscr{L}^{-1} \hspace{-0.10cm}: \ & Y(s) = \frac{\frac{5}{2}}{s}+\frac{-2}{s-3}+\frac{\frac{11}{10}}{s-4}+\frac{-\frac{3}{5}}{s+1} \\
                                         & y(t) = \frac{5}{2} \mathscr{L}^{-1} \left\{\frac{1}{s} \right\} - 2\mathscr{L}^{-1} \left\{\frac{1}{s-3} \right\} + \frac{11}{10} \mathscr{L}^{-1} \left\{\frac{1}{s-4} \right\} - \frac{3}{5} \mathscr{L}^{-1} \left\{\frac{1}{s+1} \right\} \\
                       \Longrightarrow   & \boxed{y(t) = \frac{5}{2}-2e^{3t}+\frac{11}{10}e^{4t}-\frac{3}{5}e^{-t}}
  \end{align*}

  \end{solution}

  \subsubsection*{Problem 7.2.46}
  \addcontentsline{toc}{subsubsection}{Problem 7.2.46}
  \vspace{-0.5cm}

  \begin{flalign*}
      & \text{Given: } y''-2y'+5y=0, \ y(0) = 1, \ y'(0) = 3 & \\
      & \text{Find:  } y(t) &
  \end{flalign*}
  \begin{solution}
  The main process to apply the Laplace transform, using \hyperlink{def:Theorem 7.1.1}{\textbf{Theorem 7.1.1}} and \hyperlink{def:Theorem 7.2.2}{\textbf{Theorem 7.2.2}}, over the entire equation,
  isolate $Y(s)$, and apply the inverse Laplace transform using \hyperlink{def:Theorem 7.2.1}{\textbf{Theorem 7.2.1}}.

  \begin{align*}
                                    & y''-2y'+5y=0 \\
    \mathscr{L} \hspace{-0.10cm}: \ & y''-2y'+5y=0 \\
                    \Longrightarrow & \left(-y'(0)-sy(0)+s^{2}Y(s)\right)-2\left(-y(0)+sY(s)\right) + 5Y(s)=0
  \end{align*}
  Rearrange and isolate $Y(s)$.
  \begin{align*}
          Y(s)\left(s^{2}-2s+5\right) &= y'(0)+y(0)\left(s-2\right) \\
          Y(s)\left(s^{2}-2s+5\right) &= \left(3\right)+\left(1\right)\left(s-2\right) \\
          Y(s)\left(s^{2}-2s+5\right) &= s+1 \\
          Y(s) &= \frac{s+1}{s^{2}-2s+5}
  \end{align*}
  Completing the square.
  \begin{align*}
    Y(s) &= \frac{s+1}{\left(s-1\right)^{2}+4} \\
    Y(s) &= \frac{s-1}{\left(s-1\right)^{2}+4}+\frac{2}{\left(s-1\right)^{2}+4}
  \end{align*}
  Apply the inverse Laplace transform alongside the translation \hyperlink{def:Theorem 7.3.1}{\textbf{Theorem 7.3.1}}.
  \begin{align*}
    \mathscr{L}^{-1} \hspace{-0.10cm}: \ & Y(s) = \frac{s-1}{\left(s-1\right)^{2}+2^{2}}+\frac{2}{\left(s-1\right)^{2}+2^{2}} \\
                                         & y(t) = e^{t}\cos\left(2t\right)+e^{t}\sin\left(2t\right) \\
                       \Longrightarrow   & \boxed{y(t) = e^{t}\left(\cos\left(2t\right)+\sin\left(2t\right)\right)}
  \end{align*}

  \end{solution}


\begin{boxedtheorem}[7.3.1] \hypertarget{def:Theorem 7.3.1}{}
\ \newline
\textbf{First Translation Theorem} \newline
  If $\mathscr{L}\{f(t)\}$ and $a$ is any real number, then

  \vspace{-0.25cm}
  \begin{equation}
    \mathscr{L}\{e^{at} \, f(t)\} = F(s-a)
  \end{equation}
  \vspace{0.5cm}
\end{boxedtheorem}
