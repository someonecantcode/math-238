\addcontentsline{toc}{subsection}{\textbf{Theorem 7.3.1}}
\begin{boxedtheorem}[7.3.1] \hypertarget{def:Theorem 7.3.1}{}
\ \newline
\textbf{First Translation Theorem} \newline
  If $\mathscr{L}\{f(t)\}$ and $a$ is any real number, then

  \vspace{-0.25cm}
  \begin{equation}
    \mathscr{L}\{e^{at} \, f(t)\} = F(s-a)
  \end{equation}

  \textbf{Inverse Form}

  \vspace{-0.25cm}
  \begin{equation}
    \mathscr{L}^{-1} \{F(s-a)\} = \mathscr{L}^{-1} \{\left. F(s) \, \right|_{s \to s-a}\} = e^{at} f(t)
  \end{equation}
  \vspace{0.15cm}

\end{boxedtheorem}

  \addcontentsline{toc}{subsubsection}{Problem 7.3.4}
  \vspace{-0.5cm}

  \begin{flalign*}
      & \text{Given: } \mathscr{L} \left\{ t^{10}e^{-7t}\right\} & \\
      & \text{Find:  } F(s) &
  \end{flalign*}
  \begin{solution}
  Using the table from \hyperlink{def:Theorem 7.1.1}{\textbf{Theorem 7.1.1}} and
  the Translation \hyperlink{def:Theorem 7.3.1}{\textbf{Theorem 7.3.1}}, we can obtain
  our answer.

  \begin{align*}
    \mathscr{L} \left\{ t^{10}e^{-7t}\right\} &= \left. \mathscr{L} \left\{ t^{10}\right\} \right|_{s \to s+7} \\
                                &= \left. \frac{10!}{s^{10+1}} \right|_{s \to s+7} \\
                                &= \frac{10!}{\left(s+7\right)^{11}}
  \end{align*}

  \begin{equation*}
    \Longrightarrow \boxed{F(s) = \frac{10!}{\left(s+7\right)^{11}}, \ t \geq 0}
  \end{equation*}

  \end{solution}

  \subsubsection*{Problem 7.3.6}
  \addcontentsline{toc}{subsubsection}{Problem 7.3.6}
  \vspace{-0.5cm}

  \begin{flalign*}
      & \text{Given: } \mathscr{L} \left\{e^{2t}\left(t-1\right)^{2} \right\} & \\
      & \text{Find:  } F(s) &
  \end{flalign*}
  \begin{solution}
  Using the table from \hyperlink{def:Theorem 7.1.1}{\textbf{Theorem 7.1.1}} and
  the Translation \hyperlink{def:Theorem 7.3.1}{\textbf{Theorem 7.3.1}}, we can obtain
  our answer.

  \begin{align*}
    \mathscr{L} \left\{ e^{2t}\left(t-1\right)^{2} \right\} &= \left. \mathscr{L} \left\{ \left(t-1\right)^{2} \right\} \right|_{s \to s-2} \\
                                &= \left. \mathscr{L} \left\{ t^{2}-2t+1 \right\} \right|_{s \to s-2} \\
                                &= \left. \left(\mathscr{L} \left\{ t^{2} \right\} -2 \mathscr{L} \left\{t \right\} + \mathscr{L} \left\{1 \right\} \right) \right|_{s \to s-2} \\
                                &= \left. \left(\frac{2!}{s^{2+1}}-2\frac{1!}{s^{1+1}}+\frac{1}{s}\right) \right|_{s \to s-2} \\
                                &= \left. \left(\frac{2!}{s^{3}}-2\frac{1!}{s^{2}}+\frac{1}{s}\right) \right|_{s \to s-2} \\
                                &= \frac{2}{\left(s-2\right)^{3}}-\frac{2}{\left(s-2\right)^{2}}+\frac{1}{s-2}
  \end{align*}

  \begin{equation*}
    \Longrightarrow \boxed{F(s) = \frac{2}{\left(s-2\right)^{3}}-\frac{2}{\left(s-2\right)^{2}}+\frac{1}{s-2}, \ t \geq 0}
  \end{equation*}

  \end{solution}

  \begin{align*}
    \mathscr{L}^{-1} \left\{\frac{2s+5}{s^{2}+6s+34} \right\} &= \mathscr{L}^{-1} \left\{\frac{2s+5}{\left(s+3\right)^{2}+25} \right\} \\
                                                          &= \mathscr{L}^{-1} \left\{\frac{2\left(s+3\right)-1}{\left(s+3\right)^{2}+25} \right\} \\
                                                          &= 2\mathscr{L}^{-1} \left\{\frac{\left(s+3\right)}{\left(s+3\right)^{2}+5^{2}} \right\} - \mathscr{L}^{-1} \left\{\frac{1}{\left(s+3\right)^{2}+5^{2}} \right\} \\

  \end{equation*}

  \end{solution}

  \subsubsection*{Problem 7.3.22}
  \addcontentsline{toc}{subsubsection}{Problem 7.3.22}
  \vspace{-0.5cm}

  \begin{flalign*}
      & \text{Given: } y'-y=1+te^{t}, \ y(0) = 0 & \\
      & \text{Find:  } y(t) &
  \end{flalign*}
  \begin{solution}
  The main process to apply the Laplace transform, using \hyperlink{def:Theorem 7.1.1}{\textbf{Theorem 7.1.1}} and \hyperlink{def:Theorem 7.2.2}{\textbf{Theorem 7.2.2}}, over the entire equation,
  isolate $Y(s)$, and apply the inverse Laplace transform using \hyperlink{def:Theorem 7.2.1}{\textbf{Theorem 7.2.1}}.

  \begin{align*}
                                    & y'-y=1+te^{t} \\
    \mathscr{L} \hspace{-0.10cm}: \ & y'-y=1+te^{t} \\
                                    & \left(y(0)+sY(s)\right)-Y(s)=\frac{1}{s}+\int_{0}^{\infty}e^{-st}\left(te^{t}\right)dt \\
                     & y(0)+\left(s-1\right)Y(s) = \frac{1}{s}+\left[\frac{t}{1-s}e^{t\left(1-s\right)}-\frac{1}{\left(1-s\right)^{2}}e^{t\left(1-s\right)}\right]_{0}^{\infty} \\
                     & y(0)+\left(s-1\right)Y(s) = \frac{1}{s}+\frac{1}{\left(1-s\right)^{2}} \\
     \Longrightarrow & \left(s-1\right)Y(s)=\frac{1}{s}+\frac{1}{\left(s-1\right)^{2}}-y(0)
  \end{align*}
  Rearrange and isolate $Y(s)$
  \begin{align*}
    \left(s-1\right) Y(s)  &= \frac{1}{s}+\frac{1}{\left(s-1\right)^{2}} - (0) \\
    Y(s)  &= \frac{1}{s\left(s-1\right)}+\frac{1}{\left(s-1\right)^{3}} \\
    Y(s)  &= \frac{A}{s}+\frac{B}{s-1} + \frac{1}{\left(s-1\right)^{3}}
  \end{align*}

  Using Heaviside cover up method for distinct roots.
  \begin{align*}
                         & A=-1, \ B=1 \\
    \Longrightarrow Y(s) &= -\frac{1}{s}+\frac{1}{s-1} + \frac{1}{\left(s-1\right)^{3}}
  \end{align*}
  Apply the inverse Laplace transform.
  \begin{align*}
     \mathscr{L}^{-1} \hspace{-0.10cm}: \ & Y(s) = -\frac{1}{s}+\frac{1}{s-1} + \frac{1}{\left(s-1\right)^{3}} \\
                                          & y(t) = - \mathscr{L}^{-1} \left\{\frac{1}{s} \right\} + \mathscr{L}^{-1} \left\{\frac{1}{s-1} \right\} 
                                                     + \mathscr{L}^{-1} \left\{\frac{1}{\left(s-1\right)^{3}} \right\}
  \end{align*}
  Using inverse translation property from \hyperlink{def:Theorem 7.3.1}{\textbf{Theorem 7.3.1}}
  and the table from \hyperlink{def:Theorem 7.2.1}{\textbf{Theorem 7.2.1}}.
  \begin{align*}
    & y(t) = - \mathscr{L}^{-1} \left\{\frac{1}{s} \right\} + \mathscr{L}^{-1} \left\{\frac{1}{s-1} \right\} 
                                                     + \mathscr{L}^{-1} \left\{\left. \frac{1}{s^{3}} \right|_{s \to s-1} \right\} \\
      & y(t) =  -1 + e^{t} + \frac{1}{2!} \mathscr{L}^{-1} \left\{\left. \frac{2!}{s^{2+1}} \right|_{s \to s-1} \right\} \\
          \Longrightarrow  & \boxed{y(t) = -1 + e^{t} + \frac{1}{2} e^{t} t^{2}}
  \end{align*}

  \end{solution}

  \subsubsection*{Problem 7.3.24}
  \addcontentsline{toc}{subsubsection}{Problem 7.3.24}
  \vspace{-0.5cm}

  \begin{flalign*}
      & \text{Given: } y''-4y'+4y=t^{3}e^{2t}, \ y(0) = 0, \ y'(0) = 0 & \\
      & \text{Find:  } y(t) &
  \end{flalign*}
  \begin{solution}
  The main process to apply the Laplace transform, using \hyperlink{def:Theorem 7.1.1}{\textbf{Theorem 7.1.1}} and \hyperlink{def:Theorem 7.2.2}{\textbf{Theorem 7.2.2}}, over the entire equation,
  isolate $Y(s)$, and apply the inverse Laplace transform using \hyperlink{def:Theorem 7.2.1}{\textbf{Theorem 7.2.1}}.

  \begin{align*}
                                    & y''-4y'+4y=t^{3}e^{2t} \\
    \mathscr{L} \hspace{-0.10cm}: \ & y''-4y'+4y=t^{3}e^{2t} \\ 
                                    & \left(\cancel{-y'(0)-sy(0)} +s^{2}Y(s)\right)-4\left(\cancel{-y(0)}+sY(s)\right)+4Y(s)= \int_{0}^{\infty}e^{-st}\left(t^{3}e^{2t}\right) \,dt \\
                     & \left(s^{2}-4s+4\right)Y(s) = \int_{0}^{\infty}t^{3}e^{-t\left(s-2\right)} \, dt \\
                     & \left(s^{2}-4s+4\right)Y(s) = \left[-\frac{t^{3}}{s-2}e^{-t\left(s-2\right)}-\frac{3t^{2}}{\left(s-2\right)^{2}}e^{-t\left(s-2\right)}-\frac{6t}{\left(s-2\right)^{3}}e^{-t\left(s-2\right)}-\frac{6}{\left(s-2\right)^{4}}e^{-t\left(s-2\right)}\right]_{0}^{\infty} \\
     \Longrightarrow & \left(s^{2}-4s+4\right)Y(s) = \frac{6}{\left(s-2\right)^{4}}
  \end{align*}
  Rearrange and isolate $Y(s)$
  \begin{align*}
    Y(s)  &= \frac{6}{\left(s^{2}-4s+4\right)\left(s-2\right)^{4}}\\
    Y(s)  &= \frac{6}{\left(s-2\right)^{6}} \\
    Y(s)  &= \frac{6}{5!}\frac{5!}{\left(s-2\right)^{5+1}}
  \end{align*}
  Using inverse translation property from \hyperlink{def:Theorem 7.3.1}{\textbf{Theorem 7.3.1}}
  and the table from \hyperlink{def:Theorem 7.2.1}{\textbf{Theorem 7.2.1}}.
  \begin{align*}
     \mathscr{L}^{-1} \hspace{-0.10cm}: \ & Y(s) = \frac{6}{5!}\frac{5!}{\left(s-2\right)^{5+1}} \\
                                          & y(t) = \frac{6}{5!} \mathscr{L}^{-1} \left\{\left. \frac{5!}{\left(s-2\right)^{5+1}} \right|_{s \to s-2} \right\} \\
                         \Longrightarrow  & \boxed{y(t) = \frac{1}{20} t^{5} e^{2t} }
  \end{align*}

  \end{solution}
